Pseudo code for schedule generation

    Create(daterange)
       for each day
          for each shift
             for each block
                employee = Checkprior()
                if valid(employee) = true
                   schedule employee
                   continue
                employee = pick()
                while valid(employee) = false
                   employee = pick()
                if no more employees
                   block = unfilled


    Checkprior(block)
       tally = 0
       for employees on schedule
          if employee scheduled during prior week
             tally ++
             check weeks before
          if tally >= 2
             return employee
       return null


    pick(block)
       return employee with greatest consecutive availability


    valid(employee)
       if available
          return true
       else
          return false
